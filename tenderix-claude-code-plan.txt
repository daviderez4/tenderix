### Database Schema Considerations

```sql
-- Core tables needed
CREATE TABLE tenders (
    id UUID PRIMARY KEY,
    tender_number VARCHAR(50),
    name TEXT,
    issuing_body TEXT,
    publication_date DATE,
    submission_deadline TIMESTAMP,
    status VARCHAR(20),
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE documents (
    id UUID PRIMARY KEY,
    tender_id UUID REFERENCES tenders(id),
    document_type VARCHAR(50),
    file_name TEXT,
    version INTEGER DEFAULT 1,
    is_original BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE requirements (
    id UUID PRIMARY KEY,
    tender_id UUID REFERENCES tenders(id),
    requirement_type VARCHAR(20), -- 'threshold' or 'scoring'
    description TEXT,
    source_document_id UUID REFERENCES documents(id),
    source_page INTEGER,
    source_section VARCHAR(50),
    source_quote TEXT,
    logic_operator VARCHAR(10), -- 'AND', 'OR', 'NESTED'
    status VARCHAR(20)
);

CREATE TABLE company_projects (
    id UUID PRIMARY KEY,
    company_id UUID,
    name TEXT,
    client TEXT,
    start_date DATE,
    end_date DATE,
    project_type VARCHAR(20),
    construction_scope NUMERIC,
    maintenance_scope NUMERIC,
    completion_type VARCHAR(20),
    technical_details JSONB
);
```
